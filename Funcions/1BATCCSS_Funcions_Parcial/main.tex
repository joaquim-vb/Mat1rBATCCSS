\begin{parts}
    \part $f(x) = x^2 - 5x + 6$
    \part $g(x) = \dfrac{x + 1}{x^2 - 4}$
    \part $h(x) = \sqrt{x - 5}$
    \part $i(x) = \sqrt[3]{x-4}$
\end{parts}

\question (1 punt) Donades les funcions $f(x) = 2x + 3$ i $g(x) = x^2 + 4$, calcula l'expressió algebraica i indica el domini de:
\begin{parts}
    \part $(f - g)(x)$
    \part $(f \cdot g)(x)$
    \part $\left(\dfrac{f}{g}\right)(x)$ 
    \part $(g \circ f)(x)$ 
\end{parts}
