\section*{3. Modelització de problemes reals}

\begin{tcolorbox}[colback=blaufosc!5, colframe=blaufosc, title=\textbf{Guia de treball: La funció exponencial}]
    La majoria de fenòmens de creixement o decreixement constant se serveixen de la fórmula:
    \[ y = k \cdot a^x \]
    On \textbf{$k$} és el valor inicial ($t=0$) i \textbf{$a$} és el factor de multiplicació.
    \begin{itemize}
        \item Si creix un \%, el factor és $a = (1 + r/100)$. Exemple: Creix un 5\% $\to a = 1,05$.
        \item Si decreix un \%, el factor és $a = (1 - r/100)$. Exemple: Baixa un 15\% $\to a = 0,85$.
    \end{itemize}
\end{tcolorbox}

\textbf{3.1. El vídeo viral.} Un vídeo comença amb 1 visualització ($t=0$). Cada hora es \textbf{duplica}.
\begin{enumerate}[label=\alph*)]
    \item Quantes visualitzacions hi ha a $t=1, 2, 3$ i $4$?
    \item Troba l'expressió algebraica $V(t)$.
    \item Quantes visualitzacions hi haurà al cap de 24 hores?
\end{enumerate}

\textbf{3.2. Capital bancari.} Ingresses 1.000 \euro{} a un \textbf{5\% d'interès anual compost}.
\begin{enumerate}[label=\alph*)]
    \item Quin capital tens a $t=0$ i $t=1$? (Recorda: multiplicar per 1,05).
    \item Calcula el capital per a $t=2$ i $t=3$.
    \item Troba l'expressió $C(t)$ i calcula els diners que tindràs al cap de 10 anys.
\end{enumerate}

\textbf{4.2. Equacions exponencials (Pistes explícites):}
\begin{enumerate}[label=\alph*)]
    \item $2^x = 32 \quad \to$ \textit{Pista: $32 = 2^5$}
    \item $3^{x+1} = 81 \quad \to$ \textit{Pista: $81 = 3^4$}
    \item $4^x = 16 \quad \to$ \textit{Pista: Escriu $4 = 2^2$ i $16 = 2^4$. Aplica $(2^2)^x = 2^{2x}$.}

\end{enumerate}
