    \vspace{0.2cm}
    \textit{Refrescant l'àlgebra per a les equacions exponencials}
\end{center}

\vspace{0.3cm}

\begin{tcolorbox}[colback=grisblavet, colframe=blaufosc, title=\textbf{Consells clau}]
    \begin{itemize}
        \item \textbf{Parèntesis:} El nombre de fora multiplica a TOTS els de dins. Compte amb els signes negatius davant d'un parèntesi!
        \item \textbf{Fraccions:} Si tens un nombre dividint a tota una expressió, passa a l'altre costat multiplicant a tot el bloc. Si n'hi ha diverses, busca el m.c.m.
    \end{itemize}
\end{tcolorbox}

\vspace{0.4cm}

    \item $2x - 5 = x + 3$

\section*{Nivell 2: Equacions amb fraccions i combinades (20 apartats)}
\begin{multicols}{2}
\begin{exercicis}
    \item $\dfrac{x}{2} = 6$
    \item $\dfrac{x}{3} + 2 = 5$
    \item $\dfrac{x-4}{2} = 3$
    \item $\dfrac{2x+1}{5} = 3$
    \item $\dfrac{x}{2} + \dfrac{x}{3} = 5$
    \item $\dfrac{x+1}{2} = \dfrac{x-1}{3}$
    \item $2 \cdot \left( \dfrac{x}{4} \right) = 10$
    \item $\dfrac{3(x-1)}{2} = 6$
    \item $x - \dfrac{x}{2} = 4$
    \item $\dfrac{2x-4}{3} = \dfrac{x+1}{2}$
    \item $\dfrac{1}{2}(x+4) = 5$
    \item $\dfrac{2}{3}(x-3) = 4$
    \item $\dfrac{x}{4} + 1 = \dfrac{x}{2} - 1$
    \item $\dfrac{3x+2}{4} = 2x - 7$
    \item $5 + \dfrac{x}{3} = 2(x-1)$
    \item $\dfrac{x+2}{5} + \dfrac{x-1}{2} = 3$
    \item $\dfrac{3(2x-1)}{4} = 9$
    \item $\dfrac{x}{6} - \dfrac{1}{3} = \dfrac{x}{2}$
    \item $\dfrac{2(x+3)}{3} = \dfrac{x+9}{2}$
    \item $x - \dfrac{x+1}{2} = 3$
\end{exercicis}
\end{multicols}
