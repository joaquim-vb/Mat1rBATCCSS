\section*{Exercici 1: Identificació teòrica}
Donades les següents dades sobre una funció $f(x)$ en el punt $x=a$, indica si la funció és \textbf{contínua} en aquest punt o si presenta una \textbf{discontinuïtat} (indica'n el tipus: evitable, de salt finit, o de salt infinit/asimptòtica).

\begin{enumerate}[label=\alph*)]
    \item $\lim_{x \to a^-} f(x) = 4$, \quad $\lim_{x \to a^+} f(x) = 4$, \quad $f(a) = 4$. Discontinuïtat  \dotfill
    \item $\lim_{x \to a^-} f(x) = -2$, \quad $\lim_{x \to a^+} f(x) = 5$, \quad $f(a) = 5$. Discontinuïtat  \dotfill
    \item $\lim_{x \to a^-} f(x) = 3$, \quad $\lim_{x \to a^+} f(x) = 3$, \quad $f(a)$ no existeix. Discontinuïtat \dotfill
    \item $\lim_{x \to a^-} f(x) = +\infty$, \quad $\lim_{x \to a^+} f(x) = -\infty$, \quad $f(a)$ no existeix. Discontinuïtat \dotfill
\end{enumerate}

\subsection*{Solucions - Exercici 1}
a) Contínua. \quad b) Discontinuïtat de salt finit (de valor 7). \quad c) Discontinuïtat evitable. \quad d) Discontinuïtat de salt infinit (asimptòtica).

\subsection*{Solucions - Exercici 2}
\textbf{2.1)} $f(2) = 3$. El límit quan $x \to 2$ val 3. Com que límit i imatge coincideixen, \textbf{és contínua}. \\
\textbf{2.2)} $g(3) = 0/0$ (no existeix). El límit és $\lim_{x \to 3} \frac{(x-3)(x+3)}{x-3} = \lim_{x \to 3} (x+3) = 6$. Hi ha límit però no imatge: \textbf{Discontinuïtat evitable}. \\
\textbf{2.3)} $h(1) = 3$. Límits laterals: per l'esquerra (ús de $x+2$) dona 3; per la dreta (ús de $5-x$) dona 4. No coincideixen, per tant no hi ha límit. \textbf{Discontinuïtat de salt finit}. \\
\textbf{2.4)} $k(4) = 2/0$ (no existeix). Límits laterals: a $4^-$ dona $-\infty$, a $4^+$ dona $+\infty$. \textbf{Discontinuïtat de salt infinit (asimptòtica)}.

\subsection*{Solucions - Exercici 3}
\textbf{3.1)} 
\begin{itemize}
    \item \textbf{En $x=0$}: $m(0) = 1$. Límits laterals: $0^-$ dona 1, $0^+$ dona 1. Com que límit (1) i imatge (1) coincideixen, \textbf{és contínua en $x=0$}.
    \item \textbf{En $x=2$}: $m(2) = 8$. Límits laterals: $2^-$ (fem $2^2+1$) dona 5, $2^+$ (fem 8) dona 8. No hi ha límit. \textbf{Discontinuïtat de salt finit en $x=2$}.
\end{itemize}

\textbf{3.2)} Funció $p(x) = \frac{x(x-4)}{(x-1)(x-4)}$
\begin{itemize}
    \item \textbf{En $x=0$}: Es pot substituir directament. $p(0) = \frac{0}{4} = 0$. El límit és 0. \textbf{Contínua}.
    \item \textbf{En $x=1$}: $p(1) = -3/0$ (no existeix). Límits laterals a 1 donen $\infty$. \textbf{Discontinuïtat de salt infinit (asimptòtica)}.
    \item \textbf{En $x=2$}: Es pot substituir directament. $p(2) = \frac{4-8}{4-10+4} = \frac{-4}{-2} = 2$. \textbf{Contínua}.
    \item \textbf{En $x=4$}: $p(4) = 0/0$ (no existeix). Factoritzem: $\lim_{x \to 4} \frac{x(x-4)}{(x-1)(x-4)} = \lim_{x \to 4} \frac{x}{x-1} = \frac{4}{3}$. Existeix límit però no imatge: \textbf{Discontinuïtat evitable}.
\end{itemize}
